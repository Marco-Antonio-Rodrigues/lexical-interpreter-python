<Exp> ::= <Term> ((MINUS | PLUS) <Term>)*
<Term> ::= <Factor> ((MUL | DIV) <Factor>)*
<Factor> ::= (PLUS | MINUS)* <Factor>
<Factor> ::= <Atom>
<Atom> ::= INT | FLOAT | LPAR <Exp> RPAR

# Terminais
PLUS é o +
MUL é o *
DIV é o /
MINUS é o -
LPAR é o (
RPAR é o )


# Tipos
INT é um inteiro qualquer
FLOAT é um float qualquer
